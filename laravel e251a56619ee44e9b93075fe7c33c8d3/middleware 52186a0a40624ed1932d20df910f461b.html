how to code in middleware

  • image
    this code defines, amra header dea akta key pass korbo. key er value ta middleware a check kora hobe. value match korle take controller e pathano hobe otherwise amra take return kore dibo with unauthorized text and 401 code
  • code
    <?php

namespace App\Http\Middleware;

use Closure;
use Illuminate\Http\Request;
use Symfony\Component\HttpFoundation\Response;

class DemoMiddleware
{
    public function handle(Request $request, Closure $next): Response
    {
        if($request->header('apikey')==='123xyz'){
            return $next($request);
        }
        return response()->json(['message'=>'Unauthorized'], 401);
    }
}
