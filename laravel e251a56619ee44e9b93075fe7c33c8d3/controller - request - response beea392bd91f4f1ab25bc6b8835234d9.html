  • upload file

    two ways to upoad files

    • store as - storage folder er moddhe upload hoy
    • move - public folder a upload hoy
    note: at a time dui folder a upload korte gele error dey
    • public directory using move method

      we can access this file using url like: 127.0.0.1:8000/uploads/imagename.jpg

      <?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;

class DemoController extends Controller
{
   function demoAction(Request $request): bool {
      $file = $request->file('photo');
      $file->move(public_path('uploads'), $file->getClientOriginalName());
     // this will create a folder named as uploads if not available inside 
		 // public folder
      return true;
   }
}
    • storage directory using move method
      <?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;

class DemoController extends Controller
{
   function demoAction(Request $request): bool {
      $file = $request->file('photo');
      $file->storeAs('uploads', $file->getClientOriginalName());
     // this will create a folder named as uploads if not available inside 
		 // storage folder's app folder.
		 // storage>app>uploads>imagename.jpg
      return true;
   }
}
